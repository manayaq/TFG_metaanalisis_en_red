Ahora debemos implementar las funciones para calcular los tamaÃ±os del efecto que se pueden utilizar con datos dicotÃ³micos. AquÃ­ se definen dos mÃ©todos para computar el Odds Ratio y el Risk Ratio respectivamente. Recordemos que estos valores se deben tomar en escala logarÃ­tmica.

Se ha considerado inSAN Moderada/Severa como evento y SAN/inSAN leve como no evento.

#FunciÃ³n para calcular el odds ratio de dos factores
calcular_odds_ratio <- function(a, b, c, d) {
  # a = ISAN_Severa_Factor1
  # b = ISAN_Leve_Factor1
  # c = ISAN_Severa_Factor2
  # d = ISAN_Leve_Factor2
  
  #FÃ³rmula odds ratio (tomando el logaritmo natural)
  OR <- log((a / b) / (c / d))
  
  #FÃ³rmula para calcular el error estÃ¡ndar del log(OR)
  se_log_OR <- sqrt(1/a + 1/b + 1/c + 1/d)
  
  return(list(TE = OR, seTE = se_log_OR))
}

calcular_risk_ratio <- function(a, b, c, d) {
  # a = ISAN_Severa_Factor1
  # b = ISAN_Leve_Factor1
  # c = ISAN_Severa_Factor2
  # d = ISAN_Leve_Factor2
  
  #FÃ³rmula risk ratio (tomando el logaritmo natural)
  RR <- log((a / (a + b)) / (c / (c + d)))
  
  #FÃ³rmula para calcular el error estÃ¡ndar del log(RR)
  se_log_RR <- sqrt((1/a) - (1/(a + b)) + (1/c) - (1/(c + d)))
  
  return(list(TE = RR, seTE = se_log_RR))
}
